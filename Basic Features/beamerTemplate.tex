\subsection{Transitions}
\begin{frame}
\frametitle{Transitions with itemize}
By using itemize you add a new item per slide
 
\begin{itemize}
 \item<1-> Text visible on slide 1
 \item<2-> Text visible on slide 2
 \item<3-> Text visible on slide 3
\end{itemize}
\end{frame}
